async function fetchStories() {
  try {
    const r = await fetch(STORIES_URL + '?cb=' + Date.now());

// ── STATS ─────────────────────────────────────────────────────────────────────
function updateStats() {
  document.getElementById('stat-total').textContent = allStories.length;

  const critical = allStories.filter(s => s.severity === 'critical').length;
  document.getElementById('stat-critical').textContent = critical;

  // Sum documented costs
  let total = 0;
  allStories.forEach(s => {
    if (!s.cost_impact) return;
    const num = parseFloat(s.cost_impact.replace(/[^0-9.]/g, ''));
    if (!isNaN(num)) total += num;
  });
  document.getElementById('stat-cost').textContent =
    total > 0 ? '$' + Math.round(total).toLocaleString() + '+' : 'Unknown';
}

function updateCounts() {

  });

function switchWall(wall) {
  activeWall = wall;

  document.querySelectorAll('.wall-tab').forEach(t => {
    t.classList.toggle('active', t.dataset.wall === wall);
  });

  document.querySelectorAll('.wall-panel').forEach(p => {
    p.classList.toggle('active', p.id === 'panel-' + wall);
  });
}

// ── RENDER STORIES ────────────────────────────────────────────────────────────
function renderWall(wall) {
  const grid     = document.getElementById('grid-' + wall);
  const stories  = allStories.filter(s => s.wall === wall);
  grid.innerHTML = '';

  if (stories.length === 0) {
    grid.innerHTML = `<div class="loading-state" style="color:var(--muted)">No stories yet on this wall. <a href="#" style="color:var(--accent);text-decoration:none;" onclick="document.querySelector('.submit-section').scrollIntoView({behavior:'smooth'});return false;">Be the first to submit one ↓</a></div>`;
    return;
  }

  stories.forEach(story => {
    const card = buildCard(story);
    grid.appendChild(card);
  });
}

function buildCard(story) {
  const card = document.createElement('div');
  card.className = 'story-card';
  card.id = 'card-' + story.id;

  const severityClass = 'severity-' + (story.severity || 'high');
  const costBadge = story.cost_impact
    ? `<span class="badge badge-cost">💸 ${escHtml(story.cost_impact)}</span>`
    : '';
  const sevBadge = story.severity === 'critical'
    ? `<span class="badge badge-critical">🔴 Critical</span>`
    : story.severity === 'high'
    ? `<span class="badge badge-high">🟡 High</span>`
    : '';

  const tags = (story.tags || []).slice(0,4).map(t =>
    `<span class="badge badge-tag">#${escHtml(t)}</span>`
  ).join('');

  card.innerHTML = `
    <div class="story-header" onclick="toggleCard('${escHtml(story.id)}')">
      <div class="story-severity">
        <div class="severity-dot ${severityClass}"></div>
      </div>
      <div class="story-meta-wrap">
        <div class="story-title">${escHtml(story.title)}</div>
        <div class="story-tldr">${escHtml(story.tldr)}</div>
        <div class="story-badges">
          <span class="badge badge-author">👤 ${escHtml(story.author || 'Anonymous')}</span>
          ${story.company_size ? `<span class="badge badge-author">🏢 ${escHtml(story.company_size)}</span>` : ''}
          ${costBadge}
          ${sevBadge}
        </div>
      </div>
      <div class="story-chevron">▶</div>
    </div>
    <div class="story-body">
      <div class="story-section">
        <div class="story-section-label label-story">📖 The Story</div>
        <div class="story-text">${escHtml(story.story).replace(/\n/g, '<br>')}</div>
      </div>
      <div class="story-section">
        <div class="story-section-label label-lesson">✅ The Lesson</div>
        <div class="lesson-box">${escHtml(story.lesson).replace(/\n/g, '<br>')}</div>
      </div>
      ${tags ? `<div class="story-tags">${tags}</div>` : ''}
    </div>`;

  return card;
}

function toggleCard(id) {
  const card = document.getElementById('card-' + id);
  if (!card) return;
  const wasOpen = card.classList.contains('open');
  // Close all others in same wall first
  card.closest('.stories-grid').querySelectorAll('.story-card.open').forEach(c => {
    c.classList.remove('open');
  });
  if (!wasOpen) card.classList.add('open');
}

function escHtml(str) {
  if (!str) return '';
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}
